# [026] Timebox 行程计划管理（Itinerary）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 timebox 域下新增"行程"对象（未来日历一次性事件），交付 4 个 action（1 Page + 3 CNUI）+ 完整 lifecycle 状态机（scheduled/in_progress/expired/cancelled/completed，**全部存储**，SM 驱动 transition）+ 读时合并到 /schedule（status 来自 DB）+ lazy reconcile 推进时间驱动的 transition（→in_progress, →expired）。

**Architecture:** 行程是 timebox 域内第 2 个对象类型（`lifecycle.itinerary` 与 `lifecycle.timebox` 并列），参 tasks 域 task+thread 多对象先例。**D2 reversal（2026-07-03）：状态全部存储**（不是原 D2=C 读时算）——镜像 OKR Cycle 模式（stored status + 时间戳 + 终态）。**时间驱动的 transition**（scheduled→in_progress 到日、→expired 过日）由 `reconcileItineraryStatuses()` 纯函数 + 页面 server component 加载时逐条 `submitDynamicIntent` 触发（**零 cron、零后台 job**）。`resolveObjectType`/`buildActionMap` 按 camelCase 约定自动路由 `createItinerary`→SM `create`/objectType `itinerary`；新增 `markInProgress` / `markExpired` 同样适用。`getRepo('timebox', 'itinerary')` 通过扩展 `createTimeboxGenericRepo` 加 `itinerary` 键实现（仿 tasks task+thread）。

**Tech Stack:** Next.js 16 / React 19 / TypeScript 5 / Drizzle 0.45 / PostgreSQL / shadcn-ui / react-big-calendar / vitest。

**Source of truth:** 设计稿 `~/.gstack/projects/walker2002-lifeware/walker-main-design-20260703-110126.md`（REVISED 2026-07-03 — D2 reversal 已被用户推翻；详见设计稿 Reviewer Concerns 段）。本计划实现其 A1/A2/A3。

**D2 reversal 与原 plan 关键差异（实现时注意）：**
- `status` 是存储列（5 态），不是读时算。
- mapper 直接读 `row.status`（无占位）。
- lifecycle 5 个 transition：create / markInProgress / markExpired（from {scheduled, in_progress}）/ cancel（from {scheduled, in_progress}）/ [027] markCompleted。
- 4 个 transition 时间戳列：inProgressAt / expiredAt / completedAt / cancelledAt。
- 终态 {expired, cancelled, completed} 不可删（SM 拒绝）。
- `reconcileItineraryStatuses()` 是新函数，取代原 `deriveItineraryStatus`。

```sql
-- [026] A1.2: itineraries 表（行程，D2 reversal: 5 态存储 + 4 transition 时间戳）
-- 镜像 OKR Cycle 模式：状态存 DB，SM 驱动 transition；时间驱动由 reconcile 懒推
BEGIN;

CREATE TABLE IF NOT EXISTS itineraries (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         uuid NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  schema_version  integer NOT NULL DEFAULT 1,
  title           text NOT NULL,
  detail          text,
  start_time      timestamptz NOT NULL,
  duration_min    integer NOT NULL,
  people          jsonb NOT NULL DEFAULT '[]',
  status          text NOT NULL DEFAULT 'scheduled'
                    CHECK (status IN ('scheduled', 'in_progress', 'expired', 'cancelled', 'completed')),
  in_progress_at  timestamptz,
  expired_at      timestamptz,
  completed_at    timestamptz,
  cancelled_at    timestamptz,
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

-- reconcile 查询索引：未终态 + 起始日 ≤ 今天
CREATE INDEX IF NOT EXISTS idx_itineraries_user_status_start
  ON itineraries(user_id, status, start_time);
-- Page 列表索引：按 user + status 筛
CREATE INDEX IF NOT EXISTS idx_itineraries_user_status
  ON itineraries(user_id, status);

COMMIT;
```

```ts
 * 走 SM 写库。零 cron、零后台 job。
 *
 * 纯函数：不 IO、不写库。按日历日（localDate）比较，与 /schedule loadDay 日界对齐。
 */
import type { Itinerary } from '@/usom/types/objects'

/** 本地日历日整数键（年*10000+月*100+日） */
function localDayKey(d: Date): number {
  return d.getFullYear() * 10000 + (d.getMonth() + 1) * 100 + d.getDate()
}

export type ReconcileAction =
  | { itineraryId: string; kind: 'needsMarkInProgress'; at: Date }
  | { itineraryId: string; kind: 'needsMarkExpired'; at: Date }

const TERMINAL: ReadonlyArray<Itinerary['status']> = ['expired', 'cancelled', 'completed']

export function reconcileItineraryStatuses(
  itineraries: ReadonlyArray<Itinerary>,
  now: Date,
): ReconcileAction[] {
  const nowDay = localDayKey(now)
  const actions: ReconcileAction[] = []

  for (const it of itineraries) {
    // 终态跳过（expired/cancelled/completed 不再推进）
    if ((TERMINAL as readonly string[]).includes(it.status)) continue

    const startDay = localDayKey(new Date(it.startTime))

    // 未来：scheduled 不变（in_progress 不可能发生在未来）
    if (nowDay < startDay) {
      // scheduled 留在 scheduled，in_progress 是不可能的（除非数据脏），跳过
      continue
    }

    // 当日：scheduled → in_progress
    if (nowDay === startDay) {
      if (it.status === 'scheduled') {
        actions.push({ itineraryId: it.id, kind: 'needsMarkInProgress', at: now })
      }
      // in_progress 当日不变
      continue
    }

    // 过日：scheduled → expired 或 in_progress → expired
    if (it.status === 'scheduled' || it.status === 'in_progress') {
      actions.push({ itineraryId: it.id, kind: 'needsMarkExpired', at: now })
    }
  }

  return actions
}
```

```ts
/**
 * @file itinerary repository
 * @brief Itinerary CRUD + range/reconcile/status update（[026] D2 reversal）
 *
 * D2 reversal 后：所有 5 态存 DB；updateStatus 完整支持 markInProgress/markExpired/cancel；
 * findNeedingReconcile 供 reconcileItineraryStatuses 配合用（实际 reconcile 在内存里用，
 * 这里只取候选行）。
 */
import { eq, and, gte, lte, isNull, inArray, sql } from 'drizzle-orm'
import * as s from '@/lib/db/schema'
import { db, type DbClient } from '@/lib/db'
import type { Itinerary, ItineraryStatus } from '@/usom/types/objects'
import type { USOM_ID, Timestamp } from '@/usom/types/primitives'
import { itineraryRowToUSOM, itineraryUSOMToRow } from './mappers/itinerary'

const NON_TERMINAL: ItineraryStatus[] = ['scheduled', 'in_progress']

export class ItineraryRepository {
  async findById(id: USOM_ID, userId: USOM_ID, tx: DbClient = db): Promise<Itinerary | null> {
    const rows = await tx.select().from(s.itineraries)
      .where(and(eq(s.itineraries.id, id as any), eq(s.itineraries.userId, userId as any)))
    return rows[0] ? itineraryRowToUSOM(rows[0]) : null
  }

  async save(it: Itinerary, userId: USOM_ID, tx: DbClient = db): Promise<void> {
    const row = itineraryUSOMToRow(it, userId)
    await tx.insert(s.itineraries).values(row).onConflictDoUpdate({
      target: s.itineraries.id,
      set: row,
    })
  }

  /** 单 UPDATE，无读后写（R-01）。透传 tx。 */
  async updateFields(
    id: USOM_ID,
    fields: Record<string, unknown>,
    userId: USOM_ID,
    tx: DbClient = db,
  ): Promise<Itinerary> {
    const setPayload: Record<string, unknown> = { ...fields, updatedAt: new Date() }
    await tx.update(s.itineraries).set(setPayload)
      .where(and(eq(s.itineraries.id, id as any), eq(s.itineraries.userId, userId as any)))
    const updated = await this.findById(id, userId, tx)
    if (!updated) throw new Error(`Itinerary ${id} not found after updateFields`)
    return updated
  }

  /** 软删：盖 cancelledAt + status='cancelled'（D2 reversal）。从 {scheduled, in_progress} 取消。 */
  async cancel(id: USOM_ID, userId: USOM_ID, tx: DbClient = db): Promise<void> {
    const now = new Date()
    await tx.update(s.itineraries)
      .set({ status: 'cancelled', cancelledAt: now, updatedAt: now })
      .where(and(eq(s.itineraries.id, id as any), eq(s.itineraries.userId, userId as any)))
  }

  /** D2 reversal: markInProgress 盖 inProgressAt + status='in_progress'（从 scheduled） */
  async markInProgress(id: USOM_ID, userId: USOM_ID, at: Date, tx: DbClient = db): Promise<void> {
    await tx.update(s.itineraries)
      .set({ status: 'in_progress', inProgressAt: at, updatedAt: at })
      .where(and(eq(s.itineraries.id, id as any), eq(s.itineraries.userId, userId as any)))
  }

  /** D2 reversal: markExpired 盖 expiredAt + status='expired'（从 {scheduled, in_progress}） */
  async markExpired(id: USOM_ID, userId: USOM_ID, at: Date, tx: DbClient = db): Promise<void> {
    await tx.update(s.itineraries)
      .set({ status: 'expired', expiredAt: at, updatedAt: at })
      .where(and(eq(s.itineraries.id, id as any), eq(s.itineraries.userId, userId as any)))
  }

  /** 范围查询：startTime 落在 [start,end] 且未取消。/schedule 读时合并用。 */
  async findByDateRange(start: Timestamp, end: Timestamp, userId: USOM_ID, tx: DbClient = db): Promise<Itinerary[]> {
    const rows = await tx.select().from(s.itineraries)
      .where(and(
        eq(s.itineraries.userId, userId as any),
        inArray(s.itineraries.status, NON_TERMINAL as any),
        gte(s.itineraries.startTime, new Date(start)),
        lte(s.itineraries.startTime, new Date(end)),
      ))
    return rows.map(itineraryRowToUSOM)
  }

  /**
   * findNeedingReconcile：D2 reversal reconcile 的 DB 候选行查询
   * 选 userId 全部 status ∈ {scheduled, in_progress} 的非终态行程，
   * 内存里 reconcileItineraryStatuses 决定是否真的需要推进。
   * 用 idx_itineraries_user_status_start 索引。
   */
  async findNeedingReconcile(userId: USOM_ID, tx: DbClient = db): Promise<Itinerary[]> {
    const rows = await tx.select().from(s.itineraries)
      .where(and(
        eq(s.itineraries.userId, userId as any),
        inArray(s.itineraries.status, NON_TERMINAL as any),
      ))
    return rows.map(itineraryRowToUSOM)
  }

  /** 列表查询：未取消 + 按 startTime 升序（行程 Page 列表视图用）。 */
  async findActive(userId: USOM_ID, tx: DbClient = db): Promise<Itinerary[]> {
    const rows = await tx.select().from(s.itineraries)
      .where(and(eq(s.itineraries.userId, userId as any), inArray(s.itineraries.status, NON_TERMINAL as any)))
      .orderBy(s.itineraries.startTime)
    return rows.map(itineraryRowToUSOM)
  }
}
```

```ts
export async function reconcileAndAdvanceItineraries(userId: USOM_ID): Promise<{ advanced: number; errors: number }> {
  const repo = new ItineraryRepository()
  const candidates = await repo.findNeedingReconcile(userId)
  const actions = reconcileItineraryStatuses(candidates, new Date())

  let advanced = 0
  let errors = 0
  for (const act of actions) {
    try {
      // [026] codex 修复 D6：用 `kind` 判别（不是 SM action 名 `action`），避免误传给 submitDynamicIntent
      // （short SM action 名 → resolveObjectType 路由错到 timebox，静默不报）
      if (act.kind === 'needsMarkInProgress') {
        await markInProgressItinerary(act.itineraryId, act.at.toISOString())
      } else {
        await markExpiredItinerary(act.itineraryId, act.at.toISOString())
      }
      advanced++
    } catch {
      errors++
    }
  }
  return { advanced, errors }
}
```

```ts
// frontend/src/domains/timebox/components/schedule-event.ts
/**
 * @file schedule-event
 * @brief /schedule 联合事件类型（[026] A3 D2 reversal）
 * itinerary.status 直接来自 DB（5 态枚举），不再 read-time。
 */
import type { TimeboxSummary, ItinerarySummary } from '@/usom/types/summaries'
import type { ItineraryStatus } from '@/usom/types/primitives'

export type ScheduleEvent =
  | { kind: 'timebox';   id: string; title: string; start: string; end: string; status: string;           source: TimeboxSummary }
  | { kind: 'itinerary'; id: string; title: string; start: string; end: string; status: ItineraryStatus; locked: boolean; source: ItinerarySummary }

export function timeboxToEvent(tb: TimeboxSummary): ScheduleEvent {
  return { kind: 'timebox', id: tb.id, title: tb.title, start: tb.startTime,
    end: tb.endTime, status: tb.status, source: tb }
}

export function itineraryToEvent(it: ItinerarySummary): ScheduleEvent {
  const end = new Date(new Date(it.startTime).getTime() + it.durationMin * 60_000).toISOString()
  // [026] D2 reversal: status 直接来自 DB
  return { kind: 'itinerary', id: it.id, title: it.title, start: it.startTime,
    end, status: it.status, locked: true, source: it }
}

export function mergeEvents(timeboxes: TimeboxSummary[], itineraries: ItinerarySummary[]): ScheduleEvent[] {
  return [...timeboxes.map(timeboxToEvent), ...itineraries.map(itineraryToEvent)]
    .sort((a, b) => new Date(a.start).getTime() - new Date(b.start).getTime())
}
```

## Performance Trade-off（plan-eng-review 2026-07-03 补充）

**N 次顺序 SM transition 的代价**：`reconcileAndAdvanceItineraries` 在每次 `/itineraries` / `/schedule` 加载时，对每个非终态行程顺序调 `submitDynamicIntent`（每次 = server action 往返 + SM + DB 写）。

- **个人 MVP 规模**（10-20 个活跃行程）：N×约 20ms = 200-400ms 页面加载延迟，**可接受**。
- **如果 N 增长**（数百个活跃行程）：成为瓶颈。**Follow-up（[P3]）**：实现 `markMultipleItineraries(actions: ReconcileAction[], userId)` 批 server action，单事务内 N 次 UPDATE，绕开 N 次 server action 往返。本 plan 不实现（[026] 是 MVP，不预设未观察到的扩展需求）。

无其他性能问题。完成 plan-eng-review 性能评审。
