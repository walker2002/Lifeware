**Goal:** 引入逻辑日（`logical_days`）一等实体，统一 habit/timebox/appointment 跨日期事件归属，并用 LDM（逻辑日相对分钟）模型让相对时间支持「次日」。

```ts
/** @file lib/logical-day/ldm.ts
 *  @brief [029] LDM（Logical-Day relative Minutes）相对时间模型：逻辑日 0 点起的分钟数，≥1440 表示次日。 */

import { tzLocalToUtcMs } from '@/lib/tz'
import type { DateOnly, Timestamp } from '@/usom/types/primitives'

/** 一天的分钟数；LDM ≥ 此值即「次日」 */
export const LDM_MIN_PER_DAY = 1440
/** LDM 合法上限（2 天） */
export const LDM_MAX = 2880

/** "HH:MM" → LDM 分钟；dayOffset=1 表示次日 */
export function parseHHMMToLdm(hhmm: string, dayOffset = 0): number {
  const [h, m] = hhmm.split(':').map(Number)
  return dayOffset * LDM_MIN_PER_DAY + h * 60 + m
}

/** LDM → "HH:MM"（不含次日标记，调用方用 ldmDayOffset 判定） */
export function ldmToHHMM(ldm: number): string {
  const within = ((ldm % LDM_MIN_PER_DAY) + LDM_MIN_PER_DAY) % LDM_MIN_PER_DAY
  const h = Math.floor(within / 60)
  const m = within % 60
  return `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}`
}

/** LDM → 日偏移（0=当日，1=次日） */
export function ldmDayOffset(ldm: number): number {
  return Math.floor(ldm / LDM_MIN_PER_DAY)
}

export function isValidLdm(ldm: number): boolean {
  return Number.isInteger(ldm) && ldm >= 0 && ldm <= LDM_MAX
}

/** dateLabel 该日 0 点（user_tz）对应的 UTC ms */
export function dateLabelToUtcMidnightMs(dayLabel: DateOnly, tz: string): number {
  const [y, mo, d] = dayLabel.split('-').map(Number)
  return tzLocalToUtcMs(y, mo, d, 0, 0, tz)
}

/** LDM + dateLabel → 绝对 UTC ms（次日自动滚入下一天） */
export function ldmToUtcMs(dayLabel: DateOnly, ldm: number, tz: string): number {
  const [y, mo, d] = dayLabel.split('-').map(Number)
  const dayOffset = ldmDayOffset(ldm)
  const within = ldm - dayOffset * LDM_MIN_PER_DAY
  const h = Math.floor(within / 60)
  const m = within % 60
  return tzLocalToUtcMs(y, mo, d + dayOffset, h, m, tz)
}

/** 便捷：LDM + dateLabel → ISO Timestamp */
export function ldmToIso(dayLabel: DateOnly, ldm: number, tz: string): Timestamp {
  return new Date(ldmToUtcMs(dayLabel, ldm, tz)).toISOString() as Timestamp
}
```

```ts
/** @file lib/logical-day/resolver.ts
 *  @brief [029] 逻辑日归属规则：显式指定日期优先，否则默认 date(startTime, user_tz)。 */

import { getUserTzYear, getUserTzMonth, getUserTzDate } from '@/lib/tz'
import type { DateOnly, Timestamp } from '@/usom/types/primitives'

/** Date（任意时刻）→ user_tz 日历日标签 YYYY-MM-DD */
export function formatDateLabel(date: Date, tz: string): DateOnly {
  const y = getUserTzYear(date, tz)
  const mo = getUserTzMonth(date, tz)
  const d = getUserTzDate(date, tz)
  return `${y}-${String(mo).padStart(2, '0')}-${String(d).padStart(2, '0')}` as DateOnly
}

/** 归属规则：explicitLabel 非空则用之，否则 date(startTime, tz)。 */
export function resolveLogicalDayLabel(args: {
  startTime: Timestamp
  explicitLabel?: DateOnly | string | null
  tz: string
}): DateOnly {
  const { startTime, explicitLabel, tz } = args
  if (explicitLabel && typeof explicitLabel === 'string' && explicitLabel.length > 0) {
    return explicitLabel as DateOnly
  }
  return formatDateLabel(new Date(startTime), tz)
}
```

```ts
/** @file domains/timebox/repository/mappers/logical-day.ts
 *  @brief [029] logical_days DB row ↔ USOM LogicalDay 双向映射。 */

import type { LogicalDay } from '@/usom/types/objects'
import type { USOM_ID, DateOnly, Timestamp } from '@/usom/types/primitives'

type LogicalDayRow = typeof import('@/lib/db/schema').logicalDays.$inferSelect

export function logicalDayRowToUSOM(row: LogicalDayRow): LogicalDay {
  return {
    id: row.id as USOM_ID,
    userId: row.userId as USOM_ID,
    dayLabel: row.dayLabel as DateOnly,
    wakeTime: row.wakeTime ? (row.wakeTime.toISOString() as Timestamp) : null,
    sleepDurationMinutes: row.sleepDurationMinutes,
    energyBaseline: row.energyBaseline,
    reviewRating: row.reviewRating,
    reviewNotes: row.reviewNotes,
    createdAt: row.createdAt.toISOString() as Timestamp,
    updatedAt: row.updatedAt.toISOString() as Timestamp,
    schemaVersion: row.schemaVersion,
  }
}

export function logicalDayUSOMToRow(it: LogicalDay): LogicalDayRow {
  return {
    id: it.id as any,
    userId: it.userId as any,
    schemaVersion: it.schemaVersion,
    dayLabel: it.dayLabel as any,
    wakeTime: it.wakeTime ? new Date(it.wakeTime) as any : null,
    sleepDurationMinutes: it.sleepDurationMinutes,
    energyBaseline: it.energyBaseline,
    reviewRating: it.reviewRating,
    reviewNotes: it.reviewNotes,
    createdAt: new Date(it.createdAt) as any,
    updatedAt: new Date(it.updatedAt) as any,
  } as LogicalDayRow
}
```

```ts
/** @file domains/timebox/repository/logical-day.ts
 *  @brief [029] LogicalDayRepository —— 按日期查找/懒建逻辑日行。 */

import { eq, and } from 'drizzle-orm'
import * as s from '@/lib/db/schema'
import { db, type DbClient } from '@/lib/db'
import { randomUUID } from 'crypto'
import type { LogicalDay } from '@/usom/types/objects'
import type { USOM_ID, DateOnly } from '@/usom/types/primitives'
import { logicalDayRowToUSOM, logicalDayUSOMToRow } from './mappers/logical-day'

export class LogicalDayRepository {
  async findById(id: USOM_ID, userId: USOM_ID, tx: DbClient = db): Promise<LogicalDay | null> {
    const rows = await tx.select().from(s.logicalDays)
      .where(and(eq(s.logicalDays.id, id as any), eq(s.logicalDays.userId, userId as any)))
    return rows[0] ? logicalDayRowToUSOM(rows[0]) : null
  }

  async findByDate(dayLabel: DateOnly, userId: USOM_ID, tx: DbClient = db): Promise<LogicalDay | null> {
    const rows = await tx.select().from(s.logicalDays)
      .where(and(eq(s.logicalDays.dayLabel, dayLabel as any), eq(s.logicalDays.userId, userId as any)))
    return rows[0] ? logicalDayRowToUSOM(rows[0]) : null
  }

  async save(it: LogicalDay, tx: DbClient = db): Promise<void> {
    await tx.insert(s.logicalDays).values(logicalDayUSOMToRow(it))
      .onConflictDoUpdate({ target: s.logicalDays.id, set: logicalDayUSOMToRow(it) })
  }

  /** 懒建：无则创建空行（仅 id/userId/dayLabel），返回 USOM。 */
  async findOrCreateByDate(dayLabel: DateOnly, userId: USOM_ID, tx: DbClient = db): Promise<LogicalDay> {
    const existing = await this.findByDate(dayLabel, userId, tx)
    if (existing) return existing
    const now = new Date().toISOString()
    const it: LogicalDay = {
      id: randomUUID() as USOM_ID,
      userId,
      dayLabel,
      wakeTime: null, sleepDurationMinutes: null, energyBaseline: null,
      reviewRating: null, reviewNotes: null,
      createdAt: now, updatedAt: now, schemaVersion: 1,
    }
    // 并发兜底：唯一索引冲突则回读
    try {
      await this.save(it, tx)
      return it
    } catch {
      const again = await this.findByDate(dayLabel, userId, tx)
      if (again) return again
      throw new Error(`LogicalDay findOrCreate failed for ${dayLabel}`)
    }
  }
}
```

```ts
/** @file domains/timebox/repository/schedule-slot.ts
 *  @brief [029] 统一 schedule 统计入口：查 v_schedule_slots 视图（timebox+appointment 合并）。 */

import { and, eq } from 'drizzle-orm'
import { db, type DbClient } from '@/lib/db'
import { sql } from 'drizzle-orm'
import type { USOM_ID, Timestamp } from '@/usom/types/primitives'

export type SlotSourceType = 'timebox' | 'appointment'
export type SlotState = 'scheduled' | 'completed' | 'cancelled'

export interface ScheduleSlot {
  id: USOM_ID
  userId: USOM_ID
  logicalDayId: USOM_ID | null
  title: string
  startTime: Timestamp
  endTime: Timestamp
  sourceType: SlotSourceType
  sourceStatus: string
  slotState: SlotState
  activityArchetypeId: USOM_ID | null
  people: string[] | null
}

export class ScheduleSlotRepository {
  async findByLogicalDay(logicalDayId: USOM_ID, userId: USOM_ID, tx: DbClient = db): Promise<ScheduleSlot[]> {
    const rows = await tx.execute(sql`
      SELECT id, user_id, logical_day_id, title, start_time, end_time,
             activity_archetype_id, source_type, source_status, slot_state, people, tags
      FROM v_schedule_slots
      WHERE user_id = ${userId} AND logical_day_id = ${logicalDayId}
      ORDER BY start_time ASC
    `)
    return (rows as any[]).map(r => ({
      id: r.id, userId: r.user_id, logicalDayId: r.logical_day_id, title: r.title,
      startTime: new Date(r.start_time).toISOString(), endTime: new Date(r.end_time).toISOString(),
      activityArchetypeId: r.activity_archetype_id, sourceType: r.source_type,
      sourceStatus: r.source_status, slotState: r.slot_state, people: r.people,
    }))
  }
}
```

```ts
/** @file scripts/backfill-logical-day.ts
 *  @brief [029] 存量 timebox/appointment 回填 logical_day_id = date(startTime, user_tz)，懒建 logical_days 行。幂等。 */

import { db } from '../frontend/src/lib/db'
import * as s from '../frontend/src/lib/db/schema'
import { eq, isNull } from 'drizzle-orm'
import { getEffectiveTimezone } from '../frontend/src/lib/timezone-config'
import { formatDateLabel } from '../frontend/src/lib/logical-day/resolver'
import { LogicalDayRepository } from '../frontend/src/domains/timebox/repository/logical-day'

async function backfill() {
  const userId = '00000000-0000-0000-0000-000000000001' as any
  const tz = await getEffectiveTimezone(userId)
  const repo = new LogicalDayRepository()

  for (const table of [s.timeboxes, s.appointments] as const) {
    const rows = await db.select().from(table).where(isNull((table as any).logicalDayId))
    for (const r of rows) {
      const label = formatDateLabel(new Date(r.startTime), tz)
      const ld = await repo.findOrCreateByDate(label, userId)
      await db.update(table).set({ logicalDayId: ld.id } as any).where(eq((table as any).id, r.id))
    }
    console.log(`[backfill] ${(rows as any[]).length} rows on ${(table as any)._[0] ?? table}`)
  }
  process.exit(0)
}
backfill().catch(e => { console.error(e); process.exit(1) })
```
