```typescript
/**
 * @file active-habits-provider
 * @brief 活跃习惯上下文提供者
 *
 * 实现 ContextProvider 接口，提供活跃习惯数据的查询能力。
 * 无 OKR 知识——仅暴露习惯基本信息，供他域（如 OKR）贡献关联时搜索。
 */

import type { ContextProvider } from '@/usom/types/process'
import type { IHabitRepository } from '@/usom/interfaces/irepository'
import type { USOM_ID } from '@/usom/types/primitives'
import type { Habit } from '@/usom/types/objects'

export class ActiveHabitsProvider implements ContextProvider {
  constructor(private readonly repo: IHabitRepository) {}

  async provide(query: string, params: Record<string, unknown>): Promise<unknown> {
    if (query !== 'active_habits') return []

    const { userId } = params as { userId: USOM_ID }
    const habits = await this.repo.findActive(userId)

    return habits.map((h: Habit) => ({
      id: h.id,
      title: h.title,
    }))
  }
}
```

创建 `frontend/src/domains/okrs/components/contribution-panel.tsx`：

```typescript
/**
 * @file contribution-panel
 * @brief KR 贡献管理面板
 *
 * 在 KR 详情中展示和管理 task/habit 贡献关联。
 * 客户端搜索（title filter），通过 ContextProvider 获取跨域数据。
 */

"use client"

import { useState, useEffect, useMemo } from "react"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Badge } from "@/components/ui/badge"
import { resolveContext } from "@/nexus/context-engine"
import { ContributionRepository } from "@/domains/okrs/repository/contribution"
import type { Contribution } from "@/usom/types/objects"
import type { USOM_ID } from "@/usom/types/primitives"

const MVP_USER_ID = "00000000-0000-0000-0000-000000000001" as USOM_ID

interface ContributionPanelProps {
  /** 目标 KR ID */
  krId: string
  /** Objective 状态（draft/active 时允许编辑关联） */
  objectiveStatus: string
  /** 变更回调（刷新父组件数据） */
  onChange: () => void
}

/** 搜索候选项 */
interface Candidate {
  id: string
  title: string
  type: "task" | "habit"
  label: string
}

export function ContributionPanel({ krId, objectiveStatus, onChange }: ContributionPanelProps) {
  const [existingContribs, setExistingContribs] = useState<Contribution[]>([])
  const [candidates, setCandidates] = useState<Candidate[]>([])
  const [searchQuery, setSearchQuery] = useState("")
  const [isLoading, setIsLoading] = useState(true)
  const [isAdding, setIsAdding] = useState(false)
  const isEditable = objectiveStatus === "draft" || objectiveStatus === "active"

  const contribRepo = useMemo(() => new ContributionRepository(), [])

  // 加载已有贡献和候选列表
  useEffect(() => {
    let cancelled = false
    async function load() {
      setIsLoading(true)
      const contribs = await contribRepo.findByKeyResult(krId, MVP_USER_ID)
      const tasks = await resolveContext("completedTasks", "completed_ids", { userId: MVP_USER_ID }) as Array<{id: string; title: string}> | null
      const habits = await resolveContext("activeHabits", "active_habits", { userId: MVP_USER_ID }) as Array<{id: string; title: string}> | null
      if (cancelled) return
      setExistingContribs(contribs)
      const all: Candidate[] = [
        ...(tasks ?? []).map((t: {id: string; title: string}) => ({ id: t.id, title: t.title, type: "task" as const, label: "📋 任务" })),
        ...(habits ?? []).map((h: {id: string; title: string}) => ({ id: h.id, title: h.title, type: "habit" as const, label: "✅ 习惯" })),
      ]
      setCandidates(all)
      setIsLoading(false)
    }
    load()
    return () => { cancelled = true }
  }, [krId, contribRepo])

  // 搜索过滤（客户端）
  const filtered = useMemo(() => {
    if (!searchQuery.trim()) return candidates.slice(0, 20)
    const q = searchQuery.toLowerCase()
    return candidates.filter(c => c.title.toLowerCase().includes(q)).slice(0, 20)
  }, [candidates, searchQuery])

  // 关联
  const handleLink = async (candidate: Candidate) => {
    setIsAdding(true)
    await contribRepo.add({
      keyResultId: krId as USOM_ID,
      contributorType: candidate.type,
      contributorId: candidate.id as USOM_ID,
    })
    const contribs = await contribRepo.findByKeyResult(krId, MVP_USER_ID)
    setExistingContribs(contribs)
    setIsAdding(false)
    onChange()
  }

  // 解除关联
  const handleUnlink = async (contribId: string) => {
    await contribRepo.remove(contribId as USOM_ID, MVP_USER_ID)
    const contribs = await contribRepo.findByKeyResult(krId, MVP_USER_ID)
    setExistingContribs(contribs)
    onChange()
  }

  // 查找候选对应的已关联状态
  const linkedIds = new Set(existingContribs.map(c => `${c.contributorType}:${c.contributorId}`))

  if (isLoading) return (
    <div className="space-y-2 py-2">
      <div className="h-4 w-1/3 rounded bg-hairline animate-pulse" />
      <div className="h-3 w-2/3 rounded bg-hairline animate-pulse" />
    </div>
  )

  return (
    <div className="space-y-3 border-t border-hairline pt-3 mt-3">
      <div className="flex items-center justify-between">
        <h4 className="text-sm font-medium text-muted-foreground">贡献来源</h4>
      </div>

      {/* 已关联列表 */}
      {existingContribs.length === 0 ? (
        <p className="text-xs text-muted-foreground">尚未关联任何任务或习惯</p>
      ) : (
        <div className="flex flex-wrap gap-1.5">
          {existingContribs.map(c => (
            <Badge key={c.id} variant="secondary" className="gap-1 text-xs">
              {c.contributorType === "task" ? "📋" : "✅"}
              {c.contributorId.slice(0, 8)}...
              {isEditable && (
                <button type="button" onClick={() => handleUnlink(c.id)}
                  className="ml-1 text-muted-foreground hover:text-destructive">&times;</button>
              )}
            </Badge>
          ))}
        </div>
      )}

      {/* 搜索添加 */}
      {isEditable && (
        <div className="space-y-2">
          <Input
            placeholder="搜索任务或习惯名称..."
            value={searchQuery}
            onChange={e => setSearchQuery(e.target.value)}
            className="h-8 text-sm"
          />
          {searchQuery.trim() && filtered.length > 0 && (
```
