`src/domains/okrs/repository/cycle.ts`:
```ts
/**
 * @file cycle
 * @brief Cycle 仓储实现（OKR 周期一级对象）
 */
import { eq, and } from 'drizzle-orm'
import { db, type DbClient } from '../../../lib/db'
import * as s from '../../../lib/db/schema'
import type { Cycle } from '../../../usom/types/objects'
import type { USOM_ID } from '../../../usom/types/primitives'
import { cycleRowToUSOM, cycleUSOMToRow } from '../../../lib/db/repositories/mappers'

export class CycleRepository {
  async findById(id: USOM_ID, userId: USOM_ID, tx: DbClient = db): Promise<Cycle | null> {
    const rows = await tx.select().from(s.cycles)
      .where(and(eq(s.cycles.id, id), eq(s.cycles.userId, userId)))
    return rows[0] ? cycleRowToUSOM(rows[0] as any) : null
  }
  async findByUserAndStatus(status: Cycle['status'], userId: USOM_ID, tx: DbClient = db): Promise<Cycle[]> {
    const rows = await tx.select().from(s.cycles)
      .where(and(eq(s.cycles.userId, userId), eq(s.cycles.status, status)))
    return rows.map(r => cycleRowToUSOM(r as any))
  }
  async save(cycle: Cycle, userId: USOM_ID, tx: DbClient = db): Promise<void> {
    const row = cycleUSOMToRow(cycle, userId)
    await tx.insert(s.cycles).values(row).onConflictDoUpdate({ target: s.cycles.id, set: row })
  }
  async updateFields(id: USOM_ID, fields: Record<string, unknown>, userId: USOM_ID, tx: DbClient = db): Promise<Cycle> {
    await tx.update(s.cycles).set({ ...fields, updatedAt: new Date() })
      .where(and(eq(s.cycles.id, id), eq(s.cycles.userId, userId)))
    const got = await this.findById(id, userId, tx)
    if (!got) throw new Error(`Cycle ${id} not found after updateFields`)
    return got
  }
}
```

 * 运行：cd frontend && npx tsx ../scripts/022-migrate-period-to-cycle.ts

import { db } from '../frontend/src/lib/db'
import * as s from '../frontend/src/lib/db/schema'
import { sql } from 'drizzle-orm'

const KEEP = ['annual', 'quarterly', 'monthly', 'semi_annual']
async function main() {
  const rows = await db.execute(sql`
    SELECT id, user_id, period_type, period_start, period_end, status
    FROM objectives WHERE cycle_id IS NULL`)   // cycle_id 在 Task 2 已可空，无需 sentinel
  const cycleKey = new Map<string, string>()
  for (const r of rows.rows as any[]) {
    const mappedType = KEEP.includes(r.period_type) ? r.period_type : 'custom'
    const key = `${r.user_id}|${r.period_start}|${r.period_end}`
    let cycleId = cycleKey.get(key)
    if (!cycleId) {
      cycleId = crypto.randomUUID()
      await db.insert(s.cycles).values({
        id: cycleId, userId: r.user_id, cycleType: mappedType as any,
        name: KEEP.includes(r.period_type) ? `${r.period_start}~${r.period_end}`
                                           : `${r.period_type}:${r.period_start}~${r.period_end}`,
        periodStart: r.period_start, periodEnd: r.period_end,
        status: deriveCycleStatus(r.status),
      }).onConflictDoUpdate({ targetOnConstraint: 'uq_cycles_user_period', set: { updatedAt: new Date() } })
      cycleKey.set(key, cycleId)
    }
    await db.update(s.objectives).set({ cycleId }).where(sql`id = ${r.id}`)
  }
  console.log(`migrated ${rows.rows.length} objectives, ${cycleKey.size} cycles`)
}

}
